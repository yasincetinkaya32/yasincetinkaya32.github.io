Proje: Ortam Kontrol Sistemi

Arduino tabanlı sıcaklık ve nem ölçüm sistemi. LCD ekranda veri gösterimi ve servo kontrol içerir. 


#include <DHT.h>

#define DHTPIN 2
#define DHTTYPE DHT11

DHT dht(DHTPIN, DHTTYPE);

void setup() {
  Serial.begin(9600);
  dht.begin();
}

void loop() {
  float temp = dht.readTemperature();
  float hum = dht.readHumidity();
  Serial.println(temp);
  delay(2000);
}
